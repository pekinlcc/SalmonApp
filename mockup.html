明白了。当前 auth.go:74 和 auth.go:131 直接 http.SetCookie 写了原始 token。我会替换成 store.Save(ctx, sessionID)，并把 token 存到我们 internal/secstore 里的加密表。改完后 cookie 里只剩一个 opaque session id。

下面是改动的核心片段：

// middleware/auth.go
func issueSession(w http.ResponseWriter, r *http.Request, uid string) error {
    sid, err := store.New(r.Context(), uid)
    if err != nil { return err }

    http.SetCookie(w, &http.Cookie{
        Name:     "sid",
        Value:    sid,           // opaque, no token
        HttpOnly: true,
        Secure:   true,
        SameSite: http.SameSiteLaxMode,
    })
    return nil
}
